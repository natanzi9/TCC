CREATE DATABASE tcc_bolado;
USE tcc_bolado;

CREATE TABLE usuarios (
    id    INT PRIMARY KEY AUTO_INCREMENT,
    nome  VARCHAR(255) NOT NULL,
    senha VARCHAR(255)
);

CREATE TABLE estoque (
    id        INT PRIMARY KEY AUTO_INCREMENT,
    iten      VARCHAR(255) NOT NULL,
    descricao VARCHAR(255),
    qtde      INT
);

CREATE TABLE saidas (
    id         INT PRIMARY KEY AUTO_INCREMENT,
    iten       VARCHAR(255) NOT NULL,
    qtde       INT,
    finalidade VARCHAR(255)
);
